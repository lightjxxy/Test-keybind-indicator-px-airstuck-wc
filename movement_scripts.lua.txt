-- ============================================================
--  MOVEMENT SCRIPTS  (combined)
--
--  Pixel Surf        (pixelSurf / unstablePixelSurf)
--    → hold LeftShift near a wall to slide along it
--
--  Airstuck          (airstuck)
--    → hold K to freeze mid-air
--
--  Wall Climb        (autoWallhop / wallClimbV2)
--    → hold R near a wall to hop up it
--    → press Space near a wall for a gecko-style reflected jump
--
--  Keybind indicators appear side-by-side when multiple
--  features are active at the same time.
-- ============================================================

local Players           = game:GetService("Players")
local RunService        = game:GetService("RunService")
local UserInputService  = game:GetService("UserInputService")
local TweenService      = game:GetService("TweenService")
local Workspace         = game:GetService("Workspace")

local player    = Players.LocalPlayer
if not player then repeat task.wait() until Players.LocalPlayer player = Players.LocalPlayer end
local camera    = Workspace.CurrentCamera
local playerGui = player:WaitForChild("PlayerGui")

-- ── Config ────────────────────────────────────────────────
local CONFIG = {
    -- Pixel Surf
    PIXELSURF_SPEED    = 32,    -- studs/s wall-slide speed
    WALL_DETECT_DIST   = 3.5,   -- raycast distance for wall detection
    -- Wall Climb
    WALLHOP_RAY_DISTANCE  = 4,
    WALLHOP_NUM_RAYS      = 12,
    WALLHOP_COOLDOWN      = 0.18,
    WALLHOP_FLICK_ANGLE   = math.rad(18),
    WALLHOP_FLICK_DURATION = 0.12,
    WALLHOP_JUMP_DELAY    = 0.02,
    WC_BOOST_HEIGHT       = 55,  -- upward velocity for autoWallhop (slider: 20–100)
}

-- ── Keybinds ──────────────────────────────────────────────
local keybinds = {
    pixelSurf   = Enum.KeyCode.LeftShift,
    airstuck    = Enum.KeyCode.K,
    autoWallhop = Enum.KeyCode.R,
    wallClimbV2 = Enum.KeyCode.Space,
}

-- ── Feature flags ─────────────────────────────────────────
local features = {
    pixelSurf         = false,  -- stable pixel surf
    unstablePixelSurf = false,  -- periodically slips down
    airstuck          = false,  -- freeze mid-air
    autoWallhop       = false,  -- hold-key wall hop
    wallClimbV2       = false,  -- gecko press-on-jump wall climb
}

-- ── State ─────────────────────────────────────────────────
local state = {
    char  = nil,
    hum   = nil,
    root  = nil,
    menuOpen = false,

    -- Pixel Surf
    keys             = { Surf = false, Airstuck = false },
    isSurfing        = false,
    surfY            = 0,
    unstableSurfTimer = 0,
    isUnstableFalling = false,

    -- Airstuck
    isAirstuckActive = false,
    airstuckBodyPos  = nil,

    -- Wall Climb
    wallhopHolding    = false,
    isWallhopFlicking = false,
    lastWallhopTime   = 0,
}

-- ── Keybind indicator layout ──────────────────────────────
--  All indicators share a single horizontal row.
--  When multiple are active they space out automatically.
local SHARED_BASE_Y  = 0.915  -- common Y (scale) for all indicators
local SLOT_SPACING   = 70     -- px between indicator centres
local SLIDE_OFFSET_Y = 0.015  -- Y slide distance on show / hide

-- Registry: keyed by indicator name; sorted by .order when displaying.
local indicators = {}  -- { [key] = { label, order, active, baseXOffset } }

local function recalcLayout()
    local active = {}
    for _, entry in pairs(indicators) do
        if entry.active then
            table.insert(active, entry)
        end
    end
    table.sort(active, function(a, b) return a.order < b.order end)

    local n = #active
    if n == 0 then return end

    local totalWidth = (n - 1) * SLOT_SPACING
    for i, entry in ipairs(active) do
        local xOff = -totalWidth / 2 + (i - 1) * SLOT_SPACING
        entry.baseXOffset = xOff
        -- Smoothly slide already-visible labels to their new X
        if entry.label.Visible then
            TweenService:Create(entry.label,
                TweenInfo.new(0.15, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
                { Position = UDim2.new(0.5, xOff, SHARED_BASE_Y, 0) }):Play()
        end
    end
end

local function setIndicator(key, isActive)
    local entry = indicators[key]
    if not entry then return end

    entry.active = isActive
    recalcLayout()

    local xOff = entry.baseXOffset or 0
    local label = entry.label

    if isActive then
        label.Visible = true
        label.TextTransparency = 1
        label.Position = UDim2.new(0.5, xOff, SHARED_BASE_Y + SLIDE_OFFSET_Y, 0)
        TweenService:Create(label,
            TweenInfo.new(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.Out),
            { TextTransparency = 0,
              Position = UDim2.new(0.5, xOff, SHARED_BASE_Y, 0) }):Play()
    else
        local tw = TweenService:Create(label,
            TweenInfo.new(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.In),
            { TextTransparency = 1,
              Position = UDim2.new(0.5, xOff, SHARED_BASE_Y + SLIDE_OFFSET_Y, 0) })
        tw:Play()
        tw.Completed:Connect(function()
            if label.TextTransparency >= 0.99 then label.Visible = false end
        end)
    end
end

local function makeLabel(text)
    local lbl                    = Instance.new("TextLabel")
    lbl.Size                     = UDim2.new(0, 160, 0, 36)
    lbl.Position                 = UDim2.new(0.5, 0, SHARED_BASE_Y, 0)
    lbl.AnchorPoint              = Vector2.new(0.5, 0.5)
    lbl.BackgroundTransparency   = 1
    lbl.FontFace                 = Font.new("rbxassetid://12187365977", Enum.FontWeight.Bold)
    lbl.Text                     = text
    lbl.TextColor3               = Color3.fromRGB(255, 255, 255)
    lbl.TextSize                 = 30
    lbl.TextStrokeTransparency   = 0
    lbl.TextStrokeColor3         = Color3.fromRGB(0, 0, 0)
    lbl.TextTransparency         = 1
    lbl.ZIndex                   = 300
    lbl.Visible                  = false
    lbl.Parent                   = screenGui
    return lbl
end

-- Create all three labels and register them (order = left → right)
indicators["px"]  = { label = makeLabel("px"),  order = 1, active = false, baseXOffset = 0 }
indicators["air"] = { label = makeLabel("air"), order = 2, active = false, baseXOffset = 0 }
indicators["wc"]  = { label = makeLabel("wc"),  order = 3, active = false, baseXOffset = 0 }

-- ── Active-state trackers (last known, for change detection) ──
local lastPx  = false
local lastAir = false
local lastWc  = false

-- ── Pixel Surf helpers ────────────────────────────────────
local function touchingWall()
    if not state.root then return false end
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = { state.char }
    params.FilterType = Enum.RaycastFilterType.Exclude
    for _, dir in ipairs({
        state.root.CFrame.LookVector,
        -state.root.CFrame.LookVector,
        state.root.CFrame.RightVector,
        -state.root.CFrame.RightVector,
    }) do
        if Workspace:Raycast(state.root.Position, dir * CONFIG.WALL_DETECT_DIST, params) then
            return true
        end
    end
    return false
end

local function getCardinalDir()
    local look = Vector3.new(camera.CFrame.LookVector.X, 0, camera.CFrame.LookVector.Z)
    if look.Magnitude == 0 then return Vector3.new(0, 0, 1) end
    look = look.Unit
    if math.abs(look.X) > math.abs(look.Z) then
        return Vector3.new(look.X > 0 and 1 or -1, 0, 0)
    else
        return Vector3.new(0, 0, look.Z > 0 and 1 or -1)
    end
end

-- ── Wall Climb helpers ────────────────────────────────────
local function castWallhopRays(root)
    if not root then return nil end
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = { state.char }
    params.FilterType = Enum.RaycastFilterType.Exclude

    local best, bestDist = nil, math.huge
    for i = 0, CONFIG.WALLHOP_NUM_RAYS - 1 do
        local angle = (i / CONFIG.WALLHOP_NUM_RAYS) * math.pi * 2
        local dir   = Vector3.new(math.cos(angle), 0, math.sin(angle))
        local result = Workspace:Raycast(root.Position, dir * CONFIG.WALLHOP_RAY_DISTANCE, params)
        if result and result.Distance < bestDist then
            best     = result
            bestDist = result.Distance
        end
    end
    return best
end

local function performWallhop(root, wallResult)
    local now = tick()
    if now - state.lastWallhopTime < CONFIG.WALLHOP_COOLDOWN then return end
    state.lastWallhopTime = now

    local vel     = root.AssemblyLinearVelocity
    local hVelVec = Vector3.new(vel.X, 0, vel.Z)
    local hSpeed  = math.max(hVelVec.Magnitude, 22)
    local wn      = Vector3.new(wallResult.Normal.X, 0, wallResult.Normal.Z)
    if wn.Magnitude > 0.01 then wn = wn.Unit end

    local reflected = hVelVec - 2 * hVelVec:Dot(wn) * wn
    local newDir    = reflected.Magnitude > 0.1 and reflected.Unit or wn

    state.isWallhopFlicking = true
    task.delay(CONFIG.WALLHOP_JUMP_DELAY, function()
        if state.hum and state.hum.Health > 0 then
            state.hum:ChangeState(Enum.HumanoidStateType.Jumping)
            root.AssemblyLinearVelocity = Vector3.new(
                newDir.X * hSpeed,
                CONFIG.WC_BOOST_HEIGHT,
                newDir.Z * hSpeed
            )
        end
        task.delay(CONFIG.WALLHOP_FLICK_DURATION, function()
            state.isWallhopFlicking = false
        end)
    end)
end

-- ── Character wiring ──────────────────────────────────────
local function onCharAdded(char)
    state.char = char
    state.hum  = char:WaitForChild("Humanoid")
    state.root = char:WaitForChild("HumanoidRootPart")
    -- Reset all movement state on respawn
    state.keys.Surf         = false
    state.keys.Airstuck     = false
    state.isSurfing         = false
    state.surfY             = 0
    state.unstableSurfTimer = 0
    state.isUnstableFalling = false
    state.isAirstuckActive  = false
    state.airstuckBodyPos   = nil
    state.wallhopHolding    = false
    state.isWallhopFlicking = false
end

if player.Character then onCharAdded(player.Character) end
player.CharacterAdded:Connect(onCharAdded)

-- ── Input ─────────────────────────────────────────────────
UserInputService.InputBegan:Connect(function(input, gp)
    if gp or state.menuOpen then return end
    local kc = input.KeyCode

    if kc == keybinds.pixelSurf then
        state.keys.Surf = true
    end

    if kc == keybinds.airstuck and features.airstuck then
        state.keys.Airstuck = true
    end

    if kc == keybinds.autoWallhop and features.autoWallhop then
        state.wallhopHolding = true
    end

    if kc == keybinds.wallClimbV2 and features.wallClimbV2 then
        if state.root and state.hum and state.hum.Health > 0 then
            local wallResult = castWallhopRays(state.root)
            if wallResult then
                local vel     = state.root.AssemblyLinearVelocity
                local hVelVec = Vector3.new(vel.X, 0, vel.Z)
                local hSpeed  = math.max(hVelVec.Magnitude, 22)
                local wn      = Vector3.new(wallResult.Normal.X, 0, wallResult.Normal.Z)
                if wn.Magnitude > 0.01 then wn = wn.Unit end
                local reflected = hVelVec - 2 * hVelVec:Dot(wn) * wn
                local newDir    = reflected.Magnitude > 0.1 and reflected.Unit or wn
                state.hum:ChangeState(Enum.HumanoidStateType.Jumping)
                state.root.AssemblyLinearVelocity = Vector3.new(
                    newDir.X * hSpeed, 52, newDir.Z * hSpeed)
            end
        end
    end
end)

UserInputService.InputEnded:Connect(function(input)
    local kc = input.KeyCode
    if kc == keybinds.pixelSurf then
        state.keys.Surf = false
        state.isSurfing = false
    end
    if kc == keybinds.airstuck then
        state.keys.Airstuck = false
    end
    if kc == keybinds.autoWallhop then
        state.wallhopHolding = false
    end
end)

-- ── Main loop (RenderStepped) ─────────────────────────────
RunService.RenderStepped:Connect(function(dt)
    -- ── Indicator state changes ──
    local curPx  = features.pixelSurf == true and state.keys.Surf == true
    local curAir = features.airstuck  == true and state.keys.Airstuck == true
    local curWc  = features.autoWallhop == true and state.wallhopHolding == true

    if curPx  ~= lastPx  then lastPx  = curPx  setIndicator("px",  curPx)  end
    if curAir ~= lastAir then lastAir = curAir setIndicator("air", curAir) end
    if curWc  ~= lastWc  then lastWc  = curWc  setIndicator("wc",  curWc)  end

    -- ── Pixel Surf movement ──
    if not state.hum or not state.root then return end

    if features.unstablePixelSurf and state.keys.Surf and touchingWall() then
        if not state.isSurfing then
            state.isSurfing         = true
            state.surfY             = state.root.Position.Y
            state.unstableSurfTimer = 0
            state.isUnstableFalling = false
        end
        state.unstableSurfTimer = state.unstableSurfTimer + dt
        local dir = getCardinalDir()
        if state.isUnstableFalling then
            if state.unstableSurfTimer > 0.25 then
                state.isUnstableFalling = false
                state.unstableSurfTimer = 0
                state.surfY = state.root.Position.Y
            else
                state.root.AssemblyLinearVelocity = Vector3.new(
                    dir.X * CONFIG.PIXELSURF_SPEED, -25, dir.Z * CONFIG.PIXELSURF_SPEED)
            end
        else
            if state.unstableSurfTimer > 1.2 then
                state.isUnstableFalling = true
                state.unstableSurfTimer = 0
            else
                state.root.AssemblyLinearVelocity = Vector3.new(
                    dir.X * CONFIG.PIXELSURF_SPEED, 0, dir.Z * CONFIG.PIXELSURF_SPEED)
                state.root.CFrame = CFrame.new(
                    state.root.Position.X, state.surfY, state.root.Position.Z)
                    * (state.root.CFrame - state.root.CFrame.Position)
            end
        end

    elseif features.pixelSurf and state.keys.Surf and touchingWall() then
        if not state.isSurfing then
            state.isSurfing = true
            state.surfY     = state.root.Position.Y
        end
        local dir = getCardinalDir()
        state.root.AssemblyLinearVelocity = Vector3.new(
            dir.X * CONFIG.PIXELSURF_SPEED, 0, dir.Z * CONFIG.PIXELSURF_SPEED)
        state.root.CFrame = CFrame.new(
            state.root.Position.X, state.surfY, state.root.Position.Z)
            * (state.root.CFrame - state.root.CFrame.Position)

    else
        state.isSurfing         = false
        state.isUnstableFalling = false
    end
end)

-- ── Heartbeat ─────────────────────────────────────────────
RunService.Heartbeat:Connect(function()
    -- Airstuck
    local shouldAirstuck = features.airstuck and state.keys.Airstuck
    if shouldAirstuck then
        if not state.isAirstuckActive then
            state.isAirstuckActive = true
            if state.root then
                local bp        = Instance.new("BodyPosition")
                bp.Name         = "AirstuckLock"
                bp.MaxForce     = Vector3.new(1e9, 1e9, 1e9)
                bp.P            = 1e5
                bp.D            = 1e4
                bp.Position     = state.root.Position
                bp.Parent       = state.root
                state.airstuckBodyPos = bp
            end
        end
        if state.root then
            state.root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        end
    else
        if state.isAirstuckActive then
            state.isAirstuckActive = false
            if state.airstuckBodyPos and state.airstuckBodyPos.Parent then
                state.airstuckBodyPos:Destroy()
            end
            state.airstuckBodyPos = nil
        end
    end

    -- autoWallhop
    if features.autoWallhop and state.wallhopHolding
        and not state.isWallhopFlicking
        and not state.menuOpen then
        if state.root and state.hum and state.hum.Health > 0 then
            local wallResult = castWallhopRays(state.root)
            if wallResult then performWallhop(state.root, wallResult) end
        end
    end
end)
